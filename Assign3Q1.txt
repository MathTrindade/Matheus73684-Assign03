public class Assign3Q1 {
    public static void main(String [] args) {
    
        char someChar = 'a';
        
        
        switch (someChar) {
            
            case 'A':
            case 'E':
            case 'I':
            case 'O':
            case 'U':
            
            System.out.println (someChar + " is consonant") ;
            break;
            default:
            
          System.out.println ( someChar + " is vowel");
        
        }
    }
}
